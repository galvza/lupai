    ```typescript
    import NicheInput from '@/components/analysis/NicheInput';
    import ConfirmationCard from '@/components/analysis/ConfirmationCard';
    import type { FlowStep, NicheInterpreted, AnalysisMode, UnderstandResponse, InputClassification } from '@/types/analysis';
    ```

    State variables:
    - `step: FlowStep` — starts at `'input'`
    - `nicheInput: string` — user's text input, starts empty
    - `mode: AnalysisMode` — starts at `'quick'`
    - `userBusinessUrl: string` — starts empty
    - `interpreted: NicheInterpreted | null` — starts null
    - `classification: InputClassification | null` — starts null
    - `followUpQuestions: string[] | null` — starts null
    - `error: string | null` — starts null
    - `showBanner: boolean` — starts true, checked against localStorage

    First-time banner logic (per D-30):
    - useEffect on mount: check `localStorage.getItem('lupai_banner_dismissed')`. If `'true'`, set `showBanner` to false.
    - Dismiss handler: set `showBanner` to false, set `localStorage.setItem('lupai_banner_dismissed', 'true')`.

    `handleUnderstand` async function:
    1. Set `step` to `'understanding'`, clear `error`
    2. `fetch('/api/analyze/understand', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nicheInput }) })`
    3. Parse response as `UnderstandResponse`
    4. If `!res.ok`: set `error` to `data.error ?? 'Erro desconhecido'`, set `step` to `'input'`, return
    5. Set `classification` to `data.classification`
    6. Switch on `data.classification`:
       - `'NONSENSE'`: set `error` to `data.error`, set `step` to `'input'`
       - `'MINIMAL'`: set `followUpQuestions` to `data.followUpQuestions`, set `error` to `'Precisamos de mais detalhes. ' + (data.followUpQuestions?.join(' ') ?? '')`, set `step` to `'input'`
       - `'URL'`, `'MEDIUM'`, `'EXCESSIVE'`: set `interpreted` to `data.interpreted ?? null`, set `step` to `'confirmation'`
    7. Catch: set `error` to `(err as Error).message`, set `step` to `'input'`

    `handleStartAnalysis` async function:
    1. If `!interpreted`: return
    2. Set `step` to `'starting'`
    3. `fetch('/api/analyze', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nicheInput, nicheInterpreted: interpreted, mode, userBusinessUrl: mode === 'complete' ? userBusinessUrl || null : null }) })`
    4. Parse response
    5. If `!res.ok`: set `error` to `data.error`, set `step` to `'error'`, return
    6. Store publicAccessToken: `localStorage.setItem('lupai_pat_' + data.analysisId, data.publicAccessToken)`
    7. Redirect: `window.location.href = data.redirectUrl`
    8. Catch: set `error` to message, set `step` to `'error'`

    `handleCorrect` function:
    - Set `step` to `'input'`, keep `interpreted` and `nicheInput` so user can re-edit

    `handleEditInterpreted` function `(field: keyof NicheInterpreted, value: string)`:
    - Set `interpreted` to `{ ...interpreted!, [field]: value }`

    `resetFlow` function:
    - Clear `error`, `interpreted`, `classification`, `followUpQuestions`, set `step` to `'input'`

    Render JSX:
    ```tsx
    return (

              Descreva seu negocio ou nicho e a LupAI vai encontrar seus concorrentes, analisar o mercado e entregar recomendacoes acionaveis.
            </p>
            <button onClick={dismissBanner} className="ml-4 text-gray-400 hover:text-gray-600 text-lg">&times;</button>
          </div>
        )}

        {/* Header */}
        <div className="text-center mb-8">
          <h1 className="text-4xl font-bold text-brand">LupAI</h1>
          <p className="mt-2 text-lg text-gray-600">Inteligencia de marketing com IA</p>
        </div>

        {/* Input section — always visible */}
        <NicheInput
          value={nicheInput}
          onChange={(v) => { setNicheInput(v); if (error) setError(null); }}
          mode={mode}
          onModeChange={setMode}
          userBusinessUrl={userBusinessUrl}
          onUserBusinessUrlChange={setUserBusinessUrl}
          onSubmit={handleUnderstand}

        {step === 'confirmation' && interpreted && classification && (

            interpreted={interpreted}
            onEdit={handleEditInterpreted}
            onConfirm={handleStartAnalysis}
            onCorrect={handleCorrect}
            loading={step === 'starting'}

        {step === 'starting' && (
    ```
